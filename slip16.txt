Q.1 Write a program to simulate Sequential (Contiguous) file allocation method.
Assume disk with n number of blocks. Give value of n as input. Randomly mark some
block as allocated and accordingly maintain the list of free blocks Write menu driver
program with menu options as mentioned below and implement each option
• Show Bit Vector
• Create New File
• Show Directory
• Exit

#include <stdio.h>
#include <string.h>
int main() {
 int n;
 printf("Enter number of disk blocks: ");
 scanf("%d", &n);
 int bit[n]; 
 char fname[10][20]; 
 int start[10], length[10]; 
 int fileCount = 0;
 for (int i = 0; i < n; i++)
 bit[i] = 0; 
int choice;
 do {
 printf("\n--- Sequential File Allocation ---\n");
 printf("1. Show Bit Vector\n");
 printf("2. Create New File\n");
 printf("3. Show Directory\n");
 printf("4. Delete File\n");
 printf("5. Exit\n");
 printf("Enter choice: ");
 scanf("%d", &choice);
 switch (choice) {
 case 1: // Show Bit Vector
 for (int i = 0; i < n; i++)
 printf("%d ", bit[i]);
 printf("\n");
 break;
 case 2: { 
 int s, len, free = 1;
 printf("Enter file name: ");
 scanf("%s", fname[fileCount]);
 printf("Enter starting block: ");
 scanf("%d", &s);
 printf("Enter number of blocks: ");  
scanf("%d", &len);
 for (int i = s; i < s + len; i++) {
 if (i >= n || bit[i] == 1) {
 free = 0;
 break;
 }
 }
 if (free) {
 for (int i = s; i < s + len; i++)
 bit[i] = 1;
 start[fileCount] = s;
 length[fileCount] = len;
 fileCount++;
 printf("File created successfully\n");
 } else {
 printf("Blocks not available\n");
 }
 break;
 }
 case 3: 
 printf("File\tStart\tLength\n");
 for (int i = 0; i < fileCount; i++)
 printf("%s\t%d\t%d\n",fname[i], start[i], length[i]);
 break;
 case 4: { 
 char del[20];
 printf("Enter file name to delete: ");
 scanf("%s", del);
 for (int i = 0; i < fileCount; i++) {
 if (strcmp(fname[i], del) == 0) {
 for (int j = start[i];
 j < start[i] + length[i]; j++)
 bit[j] = 0;
 strcpy(fname[i], fname[fileCount - 1]);
 start[i] = start[fileCount - 1];
 length[i] = length[fileCount - 1];
 fileCount--;
 printf("File deleted\n");
 break;
 }
 }
 break;
 }
 case 5: 
printf("Exiting...\n");
 break;
 default:
 printf("Invalid choice\n");
 }
 } while (choice != 5);
 return 0;
}

Q.2 Write an MPI program to find the min number from randomly generated 1000
numbers (stored in array) on a cluster (Hint: Use MPI_Reduce)

#include <stdio.h>
#include <stdlib.h>
#include <mpi.h>
#define N 1000
int main(int argc, char *argv[]) {
int rank, size;
int numbers[N];
int local_n;
int local_min, local_max;
int global_min, global_max;
MPI_Init(&argc, &argv); 
MPI_Comm_rank(MPI_COMM_WORLD, &rank);
MPI_Comm_size(MPI_COMM_WORLD, &size);
local_n = N / size;
if(rank == 0) {
for(int i = 0; i < N; i++) {
numbers[i] = rand() % 1000;
}
}
int local_numbers[local_n];
MPI_Scatter(numbers, local_n, MPI_INT,
local_numbers, local_n, MPI_INT,
0, MPI_COMM_WORLD);
local_min = local_numbers[0];
local_max = local_numbers[0];
for(int i = 1; i < local_n; i++) {
if(local_numbers[i] < local_min)
local_min = local_numbers[i];
if(local_numbers[i] > local_max)
local_max = local_numbers[i];
}
MPI_Reduce(&local_min, &global_min, 1,
MPI_INT, MPI_MIN, 0, MPI_COMM_WORLD);
MPI_Reduce(&local_max, &global_max, 1,
MPI_INT, MPI_MAX, 0, MPI_COMM_WORLD);
if(rank == 0) {
printf("Minimum number = %d\n", global_min); 
printf("Maximum number = %d\n", global_max);
}
MPI_Finalize();
return 0;
}
