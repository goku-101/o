int Avail[10];
void accept_matrix(int A[10][10])
{
 int i, j;

 scanf("%d", &A[i][j]);
}

{
int i;
 for (i = 0; i < m; i++)
 scanf("%d", &Avail[i]);
}

{
 int i, j;
 for (i = 0; i < n; i++)
 for (j = 0; j < m; j++)
 Need[i][j] = Max[i][j] - Alloc[i][j];
}

{

 printf("\n");
 }
 printf("\nAvailable:\n");
 for (j = 0; j < m; j++)

Q.2 Write a simulation program for disk scheduling using SCAN algorithm. Accept total
number of disk blocks, disk request string, and current head position from the user.
Display the list of request in the order in which it is served. Also display the total
number of head moments.
86, 147, 91, 170, 95, 130, 102, 70
Starting Head position= 125
Direction: Left

#include <stdio.h>
#include <stdlib.h>
int main() {
int n, head, direction, totalMovement = 0;
printf("Enter the number of requests: ");
scanf("%d", &n);
int requests[n + 1]; 
int maxTrack;
printf("Enter the maximum track number: ");
scanf("%d", &maxTrack);
printf("Enter the requests (disk track numbers): ");
for (int i = 0; i < n; i++) {
scanf("%d", &requests[i]);
}
printf("Enter the initial head position: ");
scanf("%d", &head);
printf("Enter the direction (0 for left, 1 for right): ");
scanf("%d", &direction);
requests[n] = head;
n++;
 for (int i = 0; i < n - 1; i++) {
 for (int j = 0; j < n - i - 1; j++) {
 if (requests[j] > requests[j + 1]) {
 int temp = requests[j];
 requests[j] = requests[j + 1]; 
requests[j + 1] = temp;
 }
 }
 }
 printf("SCAN Disk Scheduling Algorithm\n");
 printf("Head movement sequence: %d", head);
 int pos;
 for (int i = 0; i < n; i++) {
 if (requests[i] == head) {
 pos = i;
 break;
 }
 }
 if (direction == 1) { 
 for (int i = pos + 1; i < n; i++) {
 totalMovement += abs(requests[i] - head);
 head = requests[i];
 printf(" -> %d", head);
 }
 if (head != maxTrack) { 
 totalMovement += abs(maxTrack - head); 
head = maxTrack;
 printf(" -> %d", head);
 }
 for (int i = pos - 1; i >= 0; i--) { 
 totalMovement += abs(requests[i] - head);
 head = requests[i];
 printf(" -> %d", head);
 }
 } else {
 for (int i = pos - 1; i >= 0; i--) {
 totalMovement += abs(requests[i] - head);
 head = requests[i];
 printf(" -> %d", head);
 }
 if (head != 0) { 
 totalMovement += head;
 head = 0;
 printf(" -> %d", head);
 }
 for (int i = pos + 1; i < n; i++) { 
 totalMovement += abs(requests[i] - head);
 head = requests[i];
 printf(" -> %d", head);
 }
 }
printf("\nTotal head movement: %d\n", totalMovement);
 return 0;
}
