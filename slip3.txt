Q.1 Write a C program to simulate Banker’s algorithm for the purpose of deadlock
avoidance. Consider the following snapshot of system, A, B, C and D is the
resource type.
Process Allocation Max Available
A B C D A B C D A B C D
P0 0 0 1 2 0 0 1 2 1 5 2 0
P1 1 0 0 0 1 7 5 0
P2 1 3 5 4 2 3 5 6
P3 0 6 3 2 0 6 5 2
P4 0 0 1 4 0 6 5 6
a)Calculate and display the content of need matrix?
b)Is the system in safe state? If display the safe sequence.

#include <stdio.h>
int n, m;
int Alloc[10][10], Max[10][10], Need[10][10];
int Avail[10], Work[10];
int Finish[10], Safe[10];
int Request[10], proc;
void accept_matrix(int A[10][10])
{
 int i, j;
 for(i = 0; i < n; i++)
 for(j = 0; j < m; j++)
 scanf("%d", &A[i][j]);
} 
void accept_available()
{
 int i;
 for(i = 0; i < m; i++)
 scanf("%d", &Avail[i]);
}
void find_need()
{
 int i, j;
 for(i = 0; i < n; i++)
 for(j = 0; j < m; j++)
 Need[i][j] = Max[i][j] - Alloc[i][j];
}
void accept_request()
{
 int i;
 printf("Enter process number: ");
 scanf("%d", &proc);
 printf("Enter request:\n");
 for(i = 0; i < m; i++)
scanf("%d", &Request[i]);
}
int safety_algo()
{
 int i, j, k, count = 0;
 for(i = 0; i < m; i++)
 Work[i] = Avail[i];
 for(i = 0; i < n; i++)
 Finish[i] = 0;
 while(count < n)
 {
 int allocated = 0;
 for(i = 0; i < n; i++)
 {
 if(Finish[i] == 0)
 {
 int canExecute = 1; 
for(j = 0; j < m; j++)
 {
 if(Need[i][j] > Work[j])
 {
 canExecute = 0;
 break;
 }
 }
 if(canExecute)
 {
 for(k = 0; k < m; k++)
 Work[k] += Alloc[i][k];
 Safe[count++] = i;
 Finish[i] = 1;
 allocated = 1;
 }
 }
 }
 if(allocated == 0)
 return 0; // Unsafe
 }
 return 1; // Safe 
}
/* Resource Request Algorithm */
void resource_request_algo()
{
 int i;
 /* Request <= Need */
 for(i = 0; i < m; i++)
 {
 if(Request[i] > Need[proc][i])
 {
 printf("Error: Request exceeds maximum need\n");
 return;
 }
 }
 for(i = 0; i < m; i++)
 {
 if(Request[i] > Avail[i])
 {
 printf("Resources not available. Process must wait\n");
 return;
 }
 for(i = 0; i < m; i++)
 {
 Avail[i] -= Request[i];
 Alloc[proc][i] += Request[i];
 Need[proc][i] -= Request[i];
 }
 if(safety_algo())
 {
 printf("\nSystem is SAFE\nSafe Sequence: ");
 for(i = 0; i < n; i++)
 printf("P%d ", Safe[i]);
 }
 else
 {
 printf("\nSystem is UNSAFE\n");
 }
}
int main()
{
 printf("Enter number of processes and resources: ");
scanf("%d %d", &n, &m);
 printf("Enter Allocation matrix:\n");
 accept_matrix(Alloc);
 printf("Enter Max matrix:\n");
 accept_matrix(Max);
 printf("Enter Available resources:\n");
 accept_available();
 find_need();
 accept_request();
 resource_request_algo();
 return 0;
}

Q.2 Write an MPI program to calculate sum and average of randomly generated 1000
numbers (stored in array) on a cluster

#include <mpi.h>
#include <stdio.h>
#include <stdlib.h>
#include <time.h>
#define SIZE 1000
int main(int argc, char* argv[]) {
 int rank, size;
 int local_sum = 0, global_sum = 0;
 double average;
 int numbers[SIZE];
 MPI_Init(&argc, &argv);
 MPI_Comm_rank(MPI_COMM_WORLD, &rank);
 MPI_Comm_size(MPI_COMM_WORLD, &size);
 int chunk_size = SIZE / size;
 int local_numbers[chunk_size];
 if (rank == 0) {
 srand(time(NULL));
 for (int i = 0; i < SIZE; i++) {
 numbers[i] = rand() % 100; // Random numbers (0-99)
 }
 }
 MPI_Scatter(numbers, chunk_size, MPI_INT, local_numbers, chunk_size, MPI_INT, 0,
MPI_COMM_WORLD); 
for (int i = 0; i < chunk_size; i++) {
 local_sum += local_numbers[i];
 }
 MPI_Reduce(&local_sum, &global_sum, 1, MPI_INT, MPI_SUM, 0,
MPI_COMM_WORLD);
 if (rank == 0) {
 average = (double)global_sum / SIZE;
 printf("Total Sum: %d\n", global_sum);
 printf("Average: %.2f\n", average);
 }
 MPI_Finalize();
 return 0;
}
