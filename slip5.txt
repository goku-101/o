Q.1 Consider a system with ‘m’ processes and ‘n’ resource types. Accept number of
instances for every resource type. For each process accept the allocation and maximum
requirement matrices. Write a program to display the contents of need matrix and to
check if the given request of a process can be granted immediately or no

#include <stdio.h>
int n, m;
int Alloc[10][10], Max[10][10], Need[10][10];
int Total[10], Avail[10], Work[10];
int Finish[10], Safe[10];
int Request[10], proc;
void accept_matrix(int A[10][10])
{
 int i, j;
 for(i = 0; i < n; i++)
 for(j = 0; j < m; j++)
 scanf("%d", &A[i][j]);
}
void accept_total()
{
int i;
 printf("Enter total instances of each resource:\n");
 for(i = 0; i < m; i++)
 scanf("%d", &Total[i]);
}
void find_available()
{
 int i, j, sum;
 for(i = 0; i < m; i++)
 {
 sum = 0;
 for(j = 0; j < n; j++)
 sum += Alloc[j][i];
 Avail[i] = Total[i] - sum;
 }
}
void find_need()
{
 int i, j;
 for(i = 0; i < n; i++)
 for(j = 0; j < m; j++)
Need[i][j] = Max[i][j] - Alloc[i][j];
}
void display_need()
{
 int i, j;
 printf("\nNeed Matrix:\n");
 for(i = 0; i < n; i++)
 {
 for(j = 0; j < m; j++)
 printf("%d ", Need[i][j]);
 printf("\n");
 }
}
void accept_request()
{
 int i;
 printf("\nEnter process number: ");
 scanf("%d", &proc);
 printf("Enter request:\n");
 for(i = 0; i < m; i++)
 scanf("%d", &Request[i]);
}
int safety_algo()
{
 int i, j, k, count = 0;
 for(i = 0; i < m; i++)
 Work[i] = Avail[i];
 for(i = 0; i < n; i++)
 Finish[i] = 0;
 while(count < n)
 {
 int allocated = 0;
 for(i = 0; i < n; i++)
 {
 if(Finish[i] == 0)
 {
 int canExecute = 1;
 for(j = 0; j < m; j++)
 {
 if(Need[i][j] > Work[j])
{
 canExecute = 0;
 break;
 }
 }
 if(canExecute)
 {
 for(k = 0; k < m; k++)
 Work[k] += Alloc[i][k];
 Safe[count++] = i;
 Finish[i] = 1;
 allocated = 1;
 }
 }
 }
 if(allocated == 0)
 return 0; 
 }
 return 1;
}
void resource_request_algo()
{
 int i;
 for(i = 0; i < m; i++)
 {
 if(Request[i] > Need[proc][i])
 {
 printf("\nError: Request exceeds Need\n");
 return;
 }
 }
 for(i = 0; i < m; i++)
 {
 if(Request[i] > Avail[i])
 {
 printf("\nResources not available. Process must wait\n");
 return;
 }
 }
 for(i = 0; i < m; i++)
{
 Avail[i] -= Request[i];
 Alloc[proc][i] += Request[i];
 Need[proc][i] -= Request[i];
 }
 if(safety_algo())
 {
 printf("\nRequest CAN be granted immediately\nSafe Sequence: ");
 for(i = 0; i < n; i++)
 printf("P%d ", Safe[i]);
 }
 else
 {
 printf("\nRequest CANNOT be granted (UNSAFE state)\n");
 }
}
int main()
{
 printf("Enter number of processes and resources: ");
 scanf("%d %d", &n, &m);
 printf("Enter Allocation matrix:\n");
accept_matrix(Alloc);
 printf("Enter Max matrix:\n");
 accept_matrix(Max);
 accept_total();
 find_available();
 find_need();
 display_need();
 accept_request();
 resource_request_algo();
 return 0;
}

Q.2 Write an MPI program to find the max number from randomly generated 1000 numbers
(stored in array) on a cluster (Hint: Use MPI_Reduce)

#include <stdio.h>
#include <stdlib.h>
#include <mpi.h>
#define N 1000
int main(int argc, char *argv[]) {
int rank, size;
int numbers[N];
int local_n;
int local_min, local_max;
int global_min, global_max;
MPI_Init(&argc, &argv);
MPI_Comm_rank(MPI_COMM_WORLD, &rank);
MPI_Comm_size(MPI_COMM_WORLD, &size);
local_n = N / size;
if(rank == 0) {
for(int i = 0; i < N; i++) {
numbers[i] = rand() % 1000;
local_min = local_numbers[0];
local_max = local_numbers[0];
for(int i = 1; i < local_n; i++) {
if(local_numbers[i] < local_min)
local_min = local_numbers[i];
if(local_numbers[i] > local_max)
local_max = local_numbers[i];
}
MPI_Reduce(&local_min, &global_min, 1,
MPI_INT, MPI_MIN, 0, MPI_COMM_WORLD);
MPI_Reduce(&local_max, &global_max, 1,
MPI_INT, MPI_MAX, 0, MPI_COMM_WORLD);
if(rank == 0) {
printf("Minimum number = %d\n", global_min);
printf("Maximum number = %d\n", global_max);
}
MPI_Finalize();
return 0;
}
