Q.1 Write an MPI program to calculate sum and average randomly generated 1000
numbers (stored in array) on a cluster.

#include <mpi.h>
#include <stdio.h>
#include <stdlib.h>
#include <time.h>
#define SIZE 1000 
int main(int argc, char* argv[]) {
 int rank, size;
 int local_sum = 0, global_sum = 0;
 double average;
 int numbers[SIZE];
 MPI_Init(&argc, &argv);
 MPI_Comm_rank(MPI_COMM_WORLD, &rank);
 MPI_Comm_size(MPI_COMM_WORLD, &size);
 int chunk_size = SIZE / size;
 int local_numbers[chunk_size];
 if (rank == 0) {
srand(time(NULL));
 for (int i = 0; i < SIZE; i++) {
 numbers[i] = rand() % 100; 
 }
 }
 MPI_Scatter(numbers, chunk_size, MPI_INT, local_numbers, chunk_size, MPI_INT, 0,
MPI_COMM_WORLD);
 for (int i = 0; i < chunk_size; i++) {
 local_sum += local_numbers[i];
 }
 MPI_Reduce(&local_sum, &global_sum, 1, MPI_INT, MPI_SUM, 0,
MPI_COMM_WORLD);
 if (rank == 0) {
 average = (double)global_sum / SIZE;
 printf("Total Sum: %d\n", global_sum);
 printf("Average: %.2f\n", average);
 }
 MPI_Finalize();
 return 0;
}

Q.2 Write a simulation program for disk scheduling using C-LOOK algorithm. Accept
total number of disk blocks, disk request string, and current head position from the user.
Display the list of request in the order in which it is served. Also display the total number
of head moments.
23, 89, 132, 42, 187, 69, 36, 55
Start Head Position: 40
Direction: Right

#include <stdio.h>
#include <stdlib.h>
int main() {
int n, head, direction, totalMovement = 0;
printf("Enter number of requests: ");
scanf("%d", &n);
int requests[n + 1];
printf("Enter the requests: ");
for(int i = 0; i < n; i++) {
scanf("%d", &requests[i]);
}
printf("Enter initial head position: ");
scanf("%d", &head);
printf("Enter direction (0 for left, 1 for right): ");
scanf("%d", &direction);
requests[n] = head;
n++; 
for(int i = 0; i < n - 1; i++) {
for(int j = 0; j < n - i - 1; j++) {
if(requests[j] > requests[j + 1]) {
int temp = requests[j];
requests[j] = requests[j + 1];
requests[j + 1] = temp;
}
}
}
printf("\nC-LOOK Disk Scheduling Algorithm\n");
printf("Head movement sequence: %d", head);
int pos;
for(int i = 0; i < n; i++) {
if(requests[i] == head) {
pos = i;
break;
}
}
if(direction == 1) { 
for(int i = pos + 1; i < n; i++) {
totalMovement += abs(requests[i] - head);
head = requests[i];
printf(" -> %d", head); 
}
for(int i = 0; i < pos; i++) {
totalMovement += abs(requests[i] - head);
head = requests[i];
printf(" -> %d", head);
}
} else { 
for(int i = pos - 1; i >= 0; i--) {
totalMovement += abs(requests[i] - head);
head = requests[i];
printf(" -> %d", head);
}
for(int i = n - 1; i > pos; i--) {
totalMovement += abs(requests[i] - head);
head = requests[i];
printf(" -> %d", head);
}
}
printf("\nTotal head movement: %d\n", totalMovement);
return 0;
}
